function App() {
  const [mainView, setMainView] = useState<'evaluation' | 'sGrade'>('evaluation');

  // State for evaluation tab
  const [jobTitle, setJobTitle] = useState<string>('');
  const [jdText, setJdText] = useState<string>('');
  const [isLoading, setIsLoading] = useState<boolean>(false);
  const [error, setError] = useState<string | null>(null);
  const [evaluationResult, setEvaluationResult] = useState<EvaluationResult | null>(null);
  const [history, setHistory] = useState<HistoryItem[]>([]);
  const [activeTab, setActiveTab] = useState<'evaluation' | 'history'>('evaluation');
  
  // State for S-Grade tab
  const [sGradeJobs, setSGradeJobs] = useState<SGradeJob[]>([]);

  const handleEvaluate = useCallback(async () => {
    if (!jobTitle.trim()) {
      setError('Please provide a job title.');
      return;
    }
    if (!jdText.trim()) {
      setError('Please upload a job description file.');
      return;
    }
    if (!process.env.API_KEY) {
        setError("API key is not configured. Please ensure the API_KEY environment variable is set.");
        return;
    }

    setIsLoading(true);
    setError(null);
    setEvaluationResult(null);
    setActiveTab('evaluation');

    try {
      const pastEvaluations = history.map(h => ({
          jobTitle: h.result.jobTitle,
          summary: h.result.summary
      }));

      const result = await evaluateJd(jobTitle, jdText, pastEvaluations);
      
      const finalResult = {...result, jobTitle: result.jobTitle || jobTitle };

      setEvaluationResult(finalResult);
      setHistory(prevHistory => [...prevHistory, { jd: jdText, result: finalResult }]);
      
    } catch (e: unknown) {
      console.error(e);
      setError(e instanceof Error ? e.message : 'An unknown error occurred during evaluation.');
    } finally {
      setIsLoading(false);
    }
  }, [jobTitle, jdText, history]);

  return (
    <div className="min-h-screen bg-[#111827] flex flex-col">
      <main className="flex-grow container mx-auto p-4 sm:p-6 lg:p-8">
        <div className="w-full max-w-7xl mx-auto">
          {/* Main Tab Navigation */}
          <div className="flex border-b border-gray-700 mb-8">
              <MainTabButton title="Đánh giá giá trị công việc" isActive={mainView === 'evaluation'} onClick={() => setMainView('evaluation')} />
              <MainTabButton title="Tra cứu S-Grade" isActive={mainView === 'sGrade'} onClick={() => setMainView('sGrade')} />
          </div>

          {/* Conditional Content */}
          {mainView === 'evaluation' && (
            <div className="w-full max-w-4xl mx-auto space-y-8">
              <JdInput
                  jobTitle={jobTitle}
                  setJobTitle={setJobTitle}
                  setJdText={setJdText}
                  onEvaluate={handleEvaluate}
                  isLoading={isLoading}
                  error={error}
                  setError={setError}
              />
              {(error || evaluationResult || isLoading) && (
                  <div className="bg-[#1f2937] p-6 rounded-2xl shadow-lg border border-gray-700 mt-8 min-h-[400px]">
                      <EvaluationDisplay
                          isLoading={isLoading && !evaluationResult}
                          error={error}
                          evaluationResult={evaluationResult}
                          history={history}
                          activeTab={activeTab}
                          setActiveTab={setActiveTab}
                      />
                  </div>
              )}
            </div>
          )}
          {mainView === 'sGrade' && <SGradeLookup jobs={sGradeJobs} setJobs={setSGradeJobs} />}
        </div>
      </main>
      <Footer />
    </div>
  );
}

export default App;

function constructPrompt(jobTitle: string, jdText: string, history: PastEvaluation[]): string {
  const factorsString = JOB_EVALUATION_FACTORS.map(factor => 
    `<factor name="${factor.name}">\n${factor.levels.map(level => 
      `  <level grade="${level.grade}">${level.description}</level>`
    ).join('\n')}\n</factor>`
  ).join('\n\n');

  let historySection = '';
  if (history.length > 0) {
    historySection = `
<previous_evaluations>
  Dưới đây là các bản tóm tắt của các JD đã được phân tích trước đó. Hãy sử dụng thông tin này để so sánh JD hiện tại.
  ${history.map(h => `
  <jd>
    <title>${h.jobTitle}</title>
    <summary>
      <scope>${h.summary.scope}</scope>
      <complexity>${h.summary.complexity}</complexity>
      <impact>${h.summary.impact}</impact>
    </summary>
  </jd>
  `).join('')}
</previous_evaluations>
    `;
  }

  const fullPrompt = `
${SYSTEM_PROMPT_CONTEXT}

<evaluation_framework>
${factorsString}
</evaluation_framework>

${historySection}

${STEPS}

${OUTPUT_FORMAT}

Bây giờ, hãy phân tích Mô tả công việc (JD) sau đây cho vị trí có tiêu đề: "${jobTitle}". Trích xuất tên vị trí chính xác từ nội dung JD.

<job_description>
${jdText}
</job_description>
`;

  return fullPrompt;
}

export const evaluateJd = async (jobTitle: string, jdText: string, history: PastEvaluation[]): Promise<EvaluationResult> => {
  const prompt = constructPrompt(jobTitle, jdText, history);

  try {
    const response = await ai.models.generateContent({
      model: 'gemini-2.5-flash',
      contents: prompt,
      config: {
        responseMimeType: 'application/json',
        responseSchema: responseSchema,
        temperature: 0.2,
      },
    });

    const responseText = response.text;
    if (!responseText) {
        throw new Error("API returned an empty response. The model may have refused to answer.");
    }

    // Sanitize response text by removing potential markdown backticks
    const sanitizedText = responseText.replace(/^```json\s*|```\s*$/g, '').trim();

    const result = JSON.parse(sanitizedText);
    return result as EvaluationResult;

  } catch (error: unknown) {
    console.error("Error calling Gemini API:", error);
    if (error instanceof Error) {
        throw new Error(`Failed to evaluate JD: ${error.message}`);
    }
    throw new Error("An unknown error occurred while communicating with the AI model.");
  }
};

const formatBytes = (bytes: number, decimals = 2) => {
    if (bytes === 0) return '0 Bytes';
    const k = 1024;
    const dm = decimals < 0 ? 0 : decimals;
    const sizes = ['Bytes', 'KB', 'MB', 'GB', 'TB'];
    const i = Math.floor(Math.log(bytes) / Math.log(k));
    return parseFloat((bytes / Math.pow(k, i)).toFixed(dm)) + ' ' + sizes[i];
}

const readFileContent = async (file: File): Promise<string> => {
    return new Promise((resolve, reject) => {
        const reader = new FileReader();
        reader.onload = async (event) => {
            try {
                if (file.type === 'application/vnd.openxmlformats-officedocument.wordprocessingml.document') {
                    const arrayBuffer = event.target?.result;
                    if (arrayBuffer instanceof ArrayBuffer) {
                        const result = await mammoth.extractRawText({ arrayBuffer });
                        resolve(result.value);
                    } else {
                        reject(new Error('Failed to read ArrayBuffer from .docx file.'));
                    }
                } else if (file.type === 'text/plain') {
                    resolve(event.target?.result as string);
                } else {
                    reject(new Error(`Unsupported file type: ${file.name}. Please upload a .docx or .txt file.`));
                }
            } catch (error) {
                const message = error instanceof Error ? error.message : "Unknown error during file processing";
                reject(new Error(`Error processing file: ${message}`));
            }
        };
        reader.onerror = (error) => reject(error);

        if (file.type === 'application/vnd.openxmlformats-officedocument.wordprocessingml.document') {
            reader.readAsArrayBuffer(file);
        } else if (file.type === 'text/plain') {
            reader.readAsText(file, 'UTF-8');
        } else {
             reject(new Error(`Unsupported file type: ${file.name}. Please upload a .docx or .txt file.`));
        }
    });
};

const JdInput: React.FC<JdInputProps> = ({ jobTitle, setJobTitle, setJdText, onEvaluate, isLoading, error, setError }) => {

  const [isReadingFile, setIsReadingFile] = useState(false);
  const fileInputRef = useRef<HTMLInputElement>(null);
  const [jdText, setInternalJdText] = useState<string>('');

  const handleFile = useCallback(async (selectedFile: File | null) => {
    if (!selectedFile) return;

    if (selectedFile.size > 200 * 1024 * 1024) { // 200MB limit
        setError("File is too large. Limit is 200MB.");
        return;
    }
    
    setIsReadingFile(true);
    setError(null);
    try {
        const text = await readFileContent(selectedFile);
        setJdText(text);
        setInternalJdText(text);
        setFile(selectedFile);
        if (!jobTitle) {
            setJobTitle(selectedFile.name.replace(/\.[^/.]+$/, ""));
        }
    } catch (e: any) {
        setError(e.message);
        setFile(null);
        setJdText('');
        setInternalJdText('');
    } finally {
        setIsReadingFile(false);
    }
  }, [setJdText, setJobTitle, jobTitle, setError]);
  
  const handleRemoveFile = () => {
    setFile(null);
    setJdText('');
    setInternalJdText('');
    if (fileInputRef.current) {
        fileInputRef.current.value = "";
    }
  }

  const onDragEnter = (e: React.DragEvent<HTMLDivElement>) => {
    e.preventDefault();
    e.stopPropagation();
    setIsDragging(true);
  };
  const onDragLeave = (e: React.DragEvent<HTMLDivElement>) => {
    e.preventDefault();
    e.stopPropagation();
    setIsDragging(false);
  };
  const onDragOver = (e: React.DragEvent<HTMLDivElement>) => {
    e.preventDefault();
    e.stopPropagation();
  };
  const onDrop = (e: React.DragEvent<HTMLDivElement>) => {
    e.preventDefault();
    e.stopPropagation();
    setIsDragging(false);
    if (e.dataTransfer.files && e.dataTransfer.files.length > 0) {
      handleFile(e.dataTransfer.files[0]);
      e.dataTransfer.clearData();
    }
  };
  const onFileSelect = (e: React.ChangeEvent<HTMLInputElement>) => {
    if (e.target.files && e.target.files.length > 0) {
        handleFile(e.target.files[0]);
    }
  };
  
  const triggerFileSelect = () => fileInputRef.current?.click();

import React, { useState, useMemo, useRef, useEffect } from 'react';
import * as XLSX from 'xlsx';
import { SGradeJob } from '../types';

const useClickOutside = (ref: React.RefObject<HTMLElement>, handler: () => void) => {
    useEffect(() => {
        const listener = (event: MouseEvent | TouchEvent) => {
            if (!ref.current || ref.current.contains(event.target as Node)) {
                return;
            }
            handler();
        };
        document.addEventListener('mousedown', listener);
        document.addEventListener('touchstart', listener);
        return () => {
            document.removeEventListener('mousedown', listener);
            document.removeEventListener('touchstart', listener);
        };
    }, [ref, handler]);
};

interface SGradeLookupProps {
    jobs: SGradeJob[];
    setJobs: (jobs: SGradeJob[]) => void;
}

const SGradeLookup: React.FC<SGradeLookupProps> = ({ jobs, setJobs }) => {
    const [searchTerm, setSearchTerm] = useState('');
    const [selectedRank, setSelectedRank] = useState('all');
    const [isFilterVisible, setIsFilterVisible] = useState(false);
    const [importError, setImportError] = useState<string | null>(null);
    const [importSuccess, setImportSuccess] = useState<string | null>(null);


    const filterRef = useRef<HTMLDivElement>(null);
    const fileInputRef = useRef<HTMLInputElement>(null);
    
    useClickOutside(filterRef, () => {
        if (isFilterVisible) {
            setIsFilterVisible(false);
        }
    });

    const ranks = useMemo(() => {
        const uniqueRanks = Array.from(new Set(jobs.map(job => job.rank)));
        uniqueRanks.sort((a, b) => {
            const numA = parseInt(a.replace('S', ''), 10);
            const numB = parseInt(b.replace('S', ''), 10);
            if (isNaN(numA) || isNaN(numB)) {
                return a.localeCompare(b); // Fallback for non-S ranks
            }
            return numA - numB;
        });
        return ['all', ...uniqueRanks];
    }, [jobs]);
    
    const filteredData = useMemo(() => {
        return jobs.filter(job => {
            const term = searchTerm.toLowerCase();
            const rankMatch = selectedRank === 'all' || job.rank === selectedRank;
            const termMatch = job.positionName.toLowerCase().includes(term) || job.vietnameseName.toLowerCase().includes(term);
            return rankMatch && termMatch;
        });
    }, [searchTerm, selectedRank, jobs]);
    
    const handleDownloadTemplate = () => {
        const header = ["positionName", "vietnameseName", "positionType", "rank"];
        const exampleRow = ["Sample Position", "Vị Trí Mẫu", "Indirect", "S7"];
        const ws = XLSX.utils.aoa_to_sheet([header, exampleRow]);
        const wb = XLSX.utils.book_new();
        XLSX.utils.book_append_sheet(wb, ws, "Template");
        XLSX.writeFile(wb, "s_grade_template.xlsx");
    };

    const triggerFileImport = () => {
        fileInputRef.current?.click();
    };

    const handleFileChange = (event: React.ChangeEvent<HTMLInputElement>) => {
        setImportError(null);
        setImportSuccess(null);
        const file = event.target.files?.[0];
        if (!file) return;

        if (!file.name.endsWith('.xlsx')) {
            setImportError('Vui lòng tải lên file có định dạng .xlsx.');
            return;
        }

        const reader = new FileReader();
        reader.onload = (e) => {
            const data = e.target?.result;
            if (!data) {
                setImportError('File rỗng hoặc không thể đọc được nội dung.');
                return;
            }
            try {
                const workbook = XLSX.read(data, { type: 'array' });
                const sheetName = workbook.SheetNames[0];
                const worksheet = workbook.Sheets[sheetName];
                const newJobsData: any[] = XLSX.utils.sheet_to_json(worksheet);

                if (newJobsData.length === 0) {
                    setImportError('File không có dữ liệu (ngoài dòng header).');
                    return;
                }
                
                const headers = Object.keys(newJobsData[0]);
                const expectedHeaders = ['positionName', 'vietnameseName', 'positionType', 'rank'];

                if (!expectedHeaders.every(h => headers.includes(h))) {
                     setImportError(`Cấu trúc file không hợp lệ. Header phải chứa các cột: ${expectedHeaders.join(', ')}`);
                     return;
                }
                
                const importedJobs: Omit<SGradeJob, 'id'>[] = newJobsData.map((row) => ({
                    positionName: String(row.positionName || '').trim(),
                    vietnameseName: String(row.vietnameseName || '').trim(),
                    positionType: String(row.positionType || '').trim(),
                    rank: String(row.rank || '').trim(),
                })).filter(job => job.positionName && job.rank);

                if (importedJobs.length === 0) {
                    setImportError('Không tìm thấy dữ liệu hợp lệ nào trong file.');
                    return;
                }

                if (jobs.length === 0) {
                    // First import, set as master data
                    const jobsWithIds = importedJobs.map((job, index) => ({...job, id: index + 1 }));
                    setJobs(jobsWithIds);
                    setImportSuccess(`Đã nhập thành công ${jobsWithIds.length} vị trí làm dữ liệu gốc.`);
                } else {
                    // Subsequent imports, append new data
                    const existingPositionNames = new Set(jobs.map(j => j.positionName.toLowerCase()));
                    let addedCount = 0;
                    let duplicateCount = 0;

                    const newJobsToAdd = importedJobs.filter(newJob => {
                        if (existingPositionNames.has(newJob.positionName.toLowerCase())) {
                            duplicateCount++;
                            return false;
                        }
                        return true;
                    });
                    
                    if(newJobsToAdd.length > 0) {
                        const maxId = Math.max(0, ...jobs.map(j => j.id));
                        const jobsWithIds = newJobsToAdd.map((job, index) => ({...job, id: maxId + index + 1 }));
                        setJobs([...jobs, ...jobsWithIds]);
                        addedCount = jobsWithIds.length;
                    }

                    let successMessage = `Hoàn tất: Đã thêm ${addedCount} vị trí mới.`;
                    if (duplicateCount > 0) {
                        successMessage += ` Bỏ qua ${duplicateCount} vị trí do trùng lặp.`;
                    }
                     if (addedCount === 0 && duplicateCount > 0) {
                        successMessage = `Hoàn tất: Không có vị trí mới nào được thêm. Bỏ qua ${duplicateCount} vị trí do trùng lặp.`;
                    }
                     if (addedCount === 0 && duplicateCount === 0) {
                        successMessage = "Tệp không chứa vị trí mới nào để thêm."
                    }
                    setImportSuccess(successMessage);
                }

            } catch (error) {
                console.error("Error parsing XLSX:", error);
                setImportError('Đã xảy ra lỗi khi xử lý file. Vui lòng kiểm tra lại định dạng và nội dung file XLSX.');
            } finally {
                if (fileInputRef.current) {
                    fileInputRef.current.value = "";
                }
            }
        };
        reader.readAsArrayBuffer(file);
    };


    return (
        <div className="bg-[#1f2937] p-6 rounded-2xl shadow-lg border border-gray-700 w-full">
            <div className="flex flex-wrap gap-4 justify-between items-center mb-6">
                <h2 className="text-2xl font-bold text-gray-100">Tra cứu S-Grade</h2>
                <div className="flex items-center space-x-2">
                    <button
                        onClick={handleDownloadTemplate}
                        className="flex items-center space-x-2 px-3 py-2 bg-gray-800 border border-gray-600 rounded-lg text-white hover:bg-gray-700 transition text-sm"
                        title="Tải file mẫu định dạng XLSX"
                    >
                        <svg xmlns="http://www.w3.org/2000/svg" className="h-5 w-5" viewBox="0 0 20 20" fill="currentColor"><path fillRule="evenodd" d="M3 17a1 1 0 011-1h12a1 1 0 110 2H4a1 1 0 01-1-1zm3.172-5.172a1 1 0 011.414 0L10 14.172l2.414-2.414a1 1 0 111.414 1.414l-3 3a1 1 0 01-1.414 0l-3-3a1 1 0 010-1.414z" clipRule="evenodd" /><path fillRule="evenodd" d="M10 2a1 1 0 011 1v10a1 1 0 11-2 0V3a1 1 0 011-1z" clipRule="evenodd" /></svg>
                        <span>Tải file mẫu</span>
                    </button>
                    <button
                        onClick={triggerFileImport}
                        className="flex items-center space-x-2 px-3 py-2 bg-emerald-600 border border-emerald-500 rounded-lg text-white hover:bg-emerald-700 transition text-sm"
                        title="Nhập dữ liệu S-Grade từ file XLSX"
                    >
                         <svg xmlns="http://www.w3.org/2000/svg" className="h-5 w-5" viewBox="0 0 20 20" fill="currentColor"><path fillRule="evenodd" d="M3 17a1 1 0 011-1h12a1 1 0 110 2H4a1 1 0 01-1-1zM6.293 6.707a1 1 0 010-1.414l3-3a1 1 0 011.414 0l3 3a1 1 0 01-1.414 1.414L11 5.414V13a1 1 0 11-2 0V5.414L7.707 6.707a1 1 0 01-1.414 0z" clipRule="evenodd" /></svg>
                        <span>{jobs.length > 0 ? 'Nhập thêm' : 'Nhập S-Grade'}</span>
                    </button>
                    <input type="file" ref={fileInputRef} onChange={handleFileChange} accept=".xlsx, application/vnd.openxmlformats-officedocument.spreadsheetml.sheet" className="hidden" />
                    <div className="relative" ref={filterRef}>
                        <button
                            onClick={() => setIsFilterVisible(!isFilterVisible)}
                            className="flex items-center space-x-2 px-4 py-2 bg-gray-800 border border-gray-600 rounded-lg text-white hover:bg-gray-700 transition"
                            aria-haspopup="true"
                            aria-expanded={isFilterVisible}

                            <span>Lọc hiển thị</span>
                            <svg xmlns="http://www.w3.org/2000/svg" className="h-5 w-5" viewBox="0 0 20 20" fill="currentColor">
                                <path fillRule="evenodd" d="M5.293 7.293a1 1 0 011.414 0L10 10.586l3.293-3.293a1 1 0 111.414 1.414l-4 4a1 1 0 01-1.414 0l-4-4a1 1 0 010-1.414z" clipRule="evenodd" />
                            </svg>
                        </button>
                        {isFilterVisible && (
                            <div className="absolute top-full right-0 mt-2 w-72 bg-[#2d3748] border border-gray-600 rounded-lg shadow-xl p-4 z-10">
                                <div className="space-y-4">
                                    <div>
                                        <label htmlFor="search-term" className="block text-sm font-medium text-gray-300 mb-1">Mã/Tên vị trí</label>
                                        <input
                                            type="text"
                                            id="search-term"
                                            value={searchTerm}
                                            onChange={(e) => setSearchTerm(e.target.value)}
                                            placeholder="Nhập mã/tên vị trí..."
                                            className="w-full px-3 py-2 bg-gray-800 border border-gray-600 rounded-md text-white placeholder-gray-400 focus:ring-2 focus:ring-emerald-500 focus:border-emerald-500"
                                        />
                                    </div>
                                    <div>
                                        <label htmlFor="rank-filter" className="block text-sm font-medium text-gray-300 mb-1">Cấp bậc</label>
                                        <select
                                            id="rank-filter"
                                            value={selectedRank}
                                            onChange={(e) => setSelectedRank(e.target.value)}
                                            className="w-full px-3 py-2 bg-gray-800 border border-gray-600 rounded-md text-white focus:ring-2 focus:ring-emerald-500 focus:border-emerald-500"
                                        >
                                            {ranks.map(rank => (
                                                <option key={rank} value={rank}>{rank === 'all' ? 'Tất cả' : rank}</option>
                                            ))}
                                        </select>
                                    </div>
                                </div>
                            </div>
                        )}
                    </div>
                </div>
            </div>
            
            {importSuccess && (
                 <div className="bg-green-900/50 text-green-300 border border-green-700 p-3 rounded-md mb-4 text-sm" role="alert">
                    {importSuccess}

            )}
            {importError && (

                    <strong>Lỗi:</strong> {importError}
                </div>
            )}

            <div className="overflow-x-auto rounded-lg border border-gray-700">
                <table className="min-w-full divide-y divide-gray-700">
                    <thead className="bg-gray-800/50">
                        <tr>
                            <th scope="col" className="px-6 py-3 text-left text-xs font-bold text-gray-400 uppercase tracking-wider">Tên vị trí</th>
                            <th scope="col" className="px-6 py-3 text-left text-xs font-bold text-gray-400 uppercase tracking-wider">Tên Tiếng Việt</th>
                            <th scope="col" className="px-6 py-3 text-left text-xs font-bold text-gray-400 uppercase tracking-wider">Loại vị trí</th>
                            <th scope="col" className="px-6 py-3 text-left text-xs font-bold text-gray-400 uppercase tracking-wider">Cấp bậc</th>

                    <tbody className="bg-[#1f2937] divide-y divide-gray-700">

                                <td colSpan={4} className="text-center py-12 text-gray-500">
